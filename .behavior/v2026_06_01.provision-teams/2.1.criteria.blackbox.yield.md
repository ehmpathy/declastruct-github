# blackbox criteria: provision github teams

## usecase.1 = team creation

given(team does not exist in github)
  when(user declares team and runs apply)
    then(team is created in github)
      sothat(org has the declared team structure)
    then(team has declared name)
    then(team has declared description)
    then(team has declared privacy)
    then(team has declared notifications)

given(team already exists in github)
  when(user declares same team and runs apply)
    then(no changes are made)
      sothat(apply is idempotent)

## usecase.2 = team update

given(team exists in github)
  when(user changes team description and runs apply)
    then(team description is updated in github)
      sothat(declared state matches github state)

given(team exists in github)
  when(user changes team privacy and runs apply)
    then(team privacy is updated in github)

given(team exists in github)
  when(user changes team notifications and runs apply)
    then(team notifications is updated in github)

## usecase.3 = team delete

given(team exists in github but not in declarations)
  when(user runs apply)
    then(team is deleted from github)
      sothat(github matches declared state)

## usecase.4 = nested teams

given(parent team exists)
  when(user declares child team with parent reference and runs apply)
    then(child team is created under parent)
      sothat(team hierarchy is established)

given(child team exists under parent)
  when(user changes parent reference and runs apply)
    then(child team moves to new parent)
      sothat(team restructures are supported)

given(child team exists under parent)
  when(user removes parent reference and runs apply)
    then(child team becomes top-level)
      sothat(teams can be promoted to root)

## usecase.5 = member onboard

given(team exists in github)
  given(user is org member)
    when(user declares membership and runs apply)
      then(user is added to team)
        sothat(onboard is code-driven)
      then(user has declared role)

given(membership already exists)
  when(user declares same membership and runs apply)
    then(no changes are made)
      sothat(apply is idempotent)

## usecase.6 = member offboard

given(membership exists in github but not in declarations)
  when(user runs apply)
    then(membership is removed from team)
      sothat(offboard is code-driven)

## usecase.7 = role changes

given(membership exists with role=member)
  when(user changes role to maintainer and runs apply)
    then(membership role is updated to maintainer)
      sothat(promotions are code-driven)

given(membership exists with role=maintainer)
  when(user changes role to member and runs apply)
    then(membership role is updated to member)
      sothat(demotions are code-driven)

## usecase.8 = error cases

given(team slug already used by different team)
  when(user declares team with conflicting slug)
    then(error is surfaced)
      sothat(user can fix conflict)

given(user is not org member)
  when(user declares membership for that user)
    then(error is surfaced)
      sothat(user knows to invite first)

given(team with secret privacy)
  when(user declares it as child of another team)
    then(error is surfaced)
      sothat(user knows secret teams cannot be nested)

given(team has only one maintainer)
  when(user removes that maintainer)
    then(error is surfaced)
      sothat(teams always have at least one maintainer)

## usecase.9 = plan preview

given(changes are declared)
  when(user runs plan)
    then(diff of current vs desired state is shown)
      sothat(user can review before apply)
    then(no changes are made to github)
      sothat(plan is safe to run)
